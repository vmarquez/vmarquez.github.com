from pandas import DataFrame
from datetime import datetime

countyUrl = "https://raw.githubusercontent.com/CSSEGISandData/COVID-19/master/csse_covid_19_data/csse_covid_19_time_series/time_series_covid19_confirmed_US.csv"

def cleanupAndMelt(counties: DataFrame) -> DataFrame: 
    #UID,iso2,iso3,code3,FIPS,Admin2,Province_State,Country_Region,Lat,Long_,Combined_Key,....
    counties.drop('UID', axis=1, inplace=True)
    counties.drop('iso2', axis=1, inplace=True)
    counties.drop('iso3', axis=1, inplace=True)
    counties.drop('code3', axis=1, inplace=True)
    #counties.drop('Admin2', axis=1, inplace=True)
    counties.drop('Country_Region', axis=1, inplace=True)
    counties.drop('Lat', axis=1, inplace=True)
    counties.drop('Long_', axis=1, inplace=True)
    counties.drop('Combined_Key', axis=1, inplace=True)
    counties.drop('FIPS', axis=1, inplace=True)

    newCounties = counties.melt(id_vars=["Province_State", "Admin2"], 
        var_name="date", 
        value_name="totalcases")
    fmt = '%Y-%m-%d'
    newCounties['date'] = newCounties['date'].map(lambda d: datetime.strptime(d, '%m/%d/%y'))
    newCounties = newCounties.rename(columns={"Province_State" :"state", "Admin2": "county"})
    return newCounties

countyCaseCounts = cleanupAndMelt(pd.read_csv(countyUrl))

from datetime import timedelta
hospital_url = "https://data.chhs.ca.gov/dataset/6882c390-b2d7-4b9a-aefa-2068cee63e47/resource/6cd8d424-dfaa-4bdd-9410-a3d656e1176e/download/covid19data.csv"

hospital_data = pd.read_csv(hospital_url, parse_dates=['Most Recent Date'])

hospital_data.drop('Total Count Confirmed', axis=1, inplace=True)
hospital_data.drop('COVID-19 Positive Patients', axis=1, inplace=True)
hospital_data.drop('Suspected COVID-19 Positive Patients', axis=1, inplace=True)
hospital_data.drop('ICU COVID-19 Suspected Patients', axis=1, inplace=True)
hospital_data = hospital_data.rename(columns = {"Total Count Deaths" : "deaths", "ICU COVID-19 Positive Patients" : "icu", "Most Recent Date" : "date"})

grouped_hospital_data = hospital_data.groupby(['date']).sum().reset_index()
grouped_hospital_data['future_date'] = grouped_hospital_data['date'].apply(lambda d: d + timedelta(days=14))

#to get daily death count opposed to total 
grouped_hospital_data['deaths'] = grouped_hospital_data['deaths'].diff()

#join on itself to in the future two weeks
merged = pd.merge(grouped_hospital_data, grouped_hospital_data, how='inner',  left_on='date', right_on='future_date')

#get percentage
merged['survival'] = (merged['deaths_x']/merged['icu_y'])*10

merged.set_index('date_y', drop=True, inplace=True)
